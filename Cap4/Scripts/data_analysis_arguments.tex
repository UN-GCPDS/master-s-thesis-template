@loop_consumer('eeg', 'aux')
def stream(self, data, topic, frame, latency):
    print(f'Incoming data #{frame}')

        case 'eeg':
            print(f'EEG{data.shape}')
        case 'aux':
            print(f'AUX{data.shape}')
    
    print(f'Topic: {topic}')
    print(f'Latency: {latency}')
